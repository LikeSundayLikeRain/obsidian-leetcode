// src/main/sectionLockExtension.ts (this plan creates the file)
import type { EditorState, Extension } from '@codemirror/state';
import type { Plugin } from 'obsidian';

/**
 * Pure helper. Line-scans state.doc for the four canonical headings and emits a flat
 * [from, to, from, to, ...] suppression-range list per CONTEXT D-03 + D-08 + D-09.
 * Returns an empty array when no canonical headings are present.
 *
 * Section semantics:
 * - ## Problem: heading line + entire body until next H2 (or EOF) — locked.
 * - ## Code: heading line locked. Fence opener line locked. Closing ``` locked.
 *   Body BETWEEN opener and closer falls through (NOT in the returned ranges).
 *   D-09 fall-through: when findCodeFence returns null, only the heading line is locked.
 * - ## Techniques: heading line locked. Body falls through.
 * - ## Notes: heading line locked. Body falls through.
 * - ## Custom Tests: NEVER added (Phase 5 D-08 + Phase 05.5 D-03).
 *
 * D-08 nuance: heading-line lock spans line.from → nextLine.from (or line.to at EOF)
 * so the trailing \n is inside the suppressed range — prevents line-delete via Backspace.
 */
export function computeLockedRanges(state: EditorState): readonly number[];

/**
 * Composed CM6 Extension = [changeFilter, atomicRanges]. The changeFilter callback:
 * 1. Reads tr.annotation(Transaction.userEvent); when it starts with 'leetcode.'
 *    returns true (bypass — Phase 5.3 chevron + future plugin dispatches stay through).
 * 2. Reads tr.startState.field(editorInfoField)?.file; null → return true (no LC note).
 * 3. Reads plugin.app.metadataCache.getFileCache(file)?.frontmatter?.['lc-slug'];
 *    non-string or empty → return true (D-06 lc-slug gate).
 * 4. Calls computeLockedRanges(tr.startState); empty → return true; else returns
 *    the readonly number[] suppression-range array (CM6 documented contract).
 *
 * The atomicRanges facet wraps the same locked ranges as a RangeSet<Decoration> so
 * arrow-key cursor motion glides past the locked regions (RESEARCH Pattern 3).
 *
 * tr.startState is mandatory (RESEARCH Pitfall 2) — never tr.state inside the filter.
 */
export function buildSectionLockExtension(plugin: Plugin): Extension;

/**
 * Re-export of the SSoT tuple from src/notes/NoteTemplate.ts (Plan 01 added it there).
 * This re-export lets the Plan 01 RED test import LOCKED_HEADINGS from this module
 * (its current `from '../../src/main/sectionLockExtension'` import path).
 *
 * Wave 1 MUST re-export — do NOT redeclare. The single source of truth is NoteTemplate.ts.
 */
export { LOCKED_HEADINGS } from '../notes/NoteTemplate';
