## src/solve/languages.ts — New Exports (Final Shape)

```typescript
/** D-04: LC langSlug → markdown-recognized fence-tag. */
export const LC_LANG_FENCE_TAG: Readonly<Record<string, string>> = {
  python3: 'python',
  python: 'python',
  java: 'java',
  cpp: 'cpp',
  c: 'cpp',
  javascript: 'javascript',
  typescript: 'typescript',
  golang: 'go',
  rust: 'rust',
};

/** D-04: Map an LC langSlug to the markdown-recognized fence-tag.
 *  Unsupported LC slugs pass through verbatim. */
export function lcSlugToFenceTag(slug: string): string {
  return LC_LANG_FENCE_TAG[slug] ?? slug;
}

/** D-10: LC langSlug → user-friendly display label for the chevron widget. */

};

/** D-04 + D-10: Chevron dropdown order for the 8 supported LC languages.

  'python3',
  'java',

  'javascript',
  'typescript',

];
```
