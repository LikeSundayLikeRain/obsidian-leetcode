```typescript
const cache = new Map<PackId, Promise<LanguageSupport>>();

export function getLanguagePack(id: PackId): Promise<LanguageSupport> {
  const existing = cache.get(id);
  if (existing) return existing;
  const promise = loadPack(id).catch((err) => {
    cache.delete(id);   // drop failed promise so retry works
    throw err;
  });
  cache.set(id, promise);
  return promise;
}
```

**Analog:** `src/solve/languages.ts` (purity style) + `src/main/python3Highlighter.ts` (silent-on-failure posture)

```typescript
// src/main/whitespaceCopyIndent.ts
//
// Phase 5.3 D-06 — whitespace-copy indent fallback for LC languages outside
// the pragmatic 8 (C#, Ruby, Swift, Kotlin, Scala, PHP, MySQL/SQL, Bash, Dart,
// Elixir, Erlang, Racket, etc.). On Enter, new line inherits previous
```

```typescript
import { indentService } from '@codemirror/language';
// eslint-disable-next-line import/no-extraneous-dependencies -- transitive peer of obsidian; external in esbuild
import type { Extension } from '@codemirror/state';
```

**Core pattern** (RESEARCH Code Example lines 826–847 — canonical, already verified):
```typescript
export const whitespaceCopyIndentExtension: Extension = indentService.of((ctx, pos) => {
  const line = ctx.state.doc.lineAt(pos);
  if (line.number === 1) return 0;
  for (let n = line.number - 1; n >= 1; n--) {
    const prev = ctx.state.doc.line(n);
    if (prev.text.trim().length === 0) continue;
    const match = /^[\t ]*/.exec(prev.text);
    const leading = match?.[0] ?? '';
    return leading.replace(/\t/g, '  ').length;
  }
  return 0;
});
```
