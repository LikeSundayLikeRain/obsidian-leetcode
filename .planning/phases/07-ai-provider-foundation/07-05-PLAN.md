    ```typescript
    // Phase 07 AIPROV-04 — once-per-provider-switch disclosure modal + shared
    // copy constant. CONTEXT.md decision D + 07-UI-SPEC §"Disclosure modal".
    // Phase 08/09/11 each append a feature-specific bullet to

    import { App, Modal, Setting } from 'obsidian';
    import type { AIProvider, ProviderConfig } from './types';
    import { prettyName } from './types';

    export const DISCLOSURE_BASE_COPY: { willSend: string[]; neverSends: string[] } = {
      willSend: [
        'Problem text (statement, examples, constraints)',
        'Your `## Code` content',
        'The last run/submit verdict and failing test (if any)',
        'Optionally your `## Notes` (only if you opt in per feature)',
      ],
      neverSends: [
        'Vault file paths outside the active note',
        'Frontmatter that does not begin with `lc-`',
        'Any other vault content',
        'Telemetry of any kind',
      ],
    };

    export class AIDisclosureModal extends Modal {
      private acknowledged = false;

      constructor(
        app: App,
        private provider: AIProvider,
        private cfg: ProviderConfig,
        private onContinue: () => void,
        private onCancel: () => void,
      ) {
        super(app);
      }

      onOpen(): void {
        const { contentEl, titleEl } = this;
        contentEl.empty();
        contentEl.addClass('leetcode-ai-disclosure');
        titleEl.setText(`Heads up: this will send data to ${prettyName(this.provider)}`);

        const baseUrlText = this.cfg.baseUrl || '(no base URL configured yet)';
        contentEl.createEl('p', {
          text: `Active provider: ${prettyName(this.provider)} — ${baseUrlText}`,
        });

        contentEl.createEl('p', { text: 'Future AI features will send:' });
        const willList = contentEl.createEl('ul');
        for (const line of DISCLOSURE_BASE_COPY.willSend) {
          willList.createEl('li', { text: line });
        }

        contentEl.createEl('p', { text: 'The plugin never sends:' });
        const neverList = contentEl.createEl('ul');
        for (const line of DISCLOSURE_BASE_COPY.neverSends) {
          neverList.createEl('li', { text: line });
        }

        new Setting(contentEl)
          .addButton((b) => b
            .setButtonText('Cancel')
            .onClick(() => {
              this.onCancel();
              this.close();
            }),
          )
          .addButton((b) => b
            .setButtonText('I understand — continue')
            .setCta()
            .onClick(() => {
              this.acknowledged = true;
              this.onContinue();
              this.close();
            }),
          );
      }

      onClose(): void {
        if (!this.acknowledged) {
          try { this.onCancel(); } catch { /* ignore */ }
        }
        this.contentEl.empty();
      }
    }
    ```

<threat_model>

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User UI click -> SettingsStore mutation | Continue button writes disclosureAcknowledged=true. Single boolean — minimal attack surface. |
| AIClient.probe / .invoke entry -> HTTP | The gate sits BEFORE the network call. Any caller of AIClient (Plan 07-04 testActiveAIConnection, future Phase 08 invokers) inherits the protection — no per-feature wiring required. |
| Esc / overlay-click on modal -> onCancel | Modal lifecycle ensures Esc-style closes are cancellation-equivalent (UI-SPEC §"Modal closeable via Esc"). Without this, an inattentive user pressing Esc could leave the gate in an indeterminate state. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-04 | Tampering / Spoofing | AIDisclosureModal lifecycle | mitigate | `onClose()` checks `acknowledged` flag and fires `onCancel()` if user closed via Esc / X / overlay-click. Continue button explicitly sets the flag BEFORE close — no race. The modal-level guard plus the plugin-level `resolved` flag make the gate idempotent. |
| T-07-03-bypass | Spoofing | AIClient.probe / .invoke | mitigate | The gate is at the AIClient boundary (not the call site), so bypass would require a future caller to skip AIClient entirely and call provider adapters directly. Plan 07-02's resolveAdapter is exported BUT is module-internal in practice (no Phase 08 plan should import it). Documented in the AIClient JSDoc + 07-RESEARCH §Security threat #5. |
| T-07-04-persist | Tampering | SettingsStore.setProviderConfig | mitigate | Plan 07-01 Task 2's `sanitizeProviderConfig` enforces strict-true for `disclosureAcknowledged`. Even if a future caller passes a malformed object, the sanitizer collapses to false — the modal will simply re-fire (fail-safe). |
