**`## Code` fence detection pattern** (REUSE SEMANTICS from analog `findCodeSectionPre` lines 28-68, but switch from DOM-walk to regex-over-`state.doc` because CM6 has no rendered DOM at decoration build time). Canonical implementation from RESEARCH.md lines 402-421:
```typescript
// Analog behavior (codeActionsPostProcessor.ts lines 37-49):
// Walk lines backward from section start; first H1/H2 encountered decides:
//   - if it is `## Code` → attach
//   - if it is any other level-1 or level-2 → bail

// New (CM6 — regex forward-scan over state.doc):
export function findCodeFence(state: EditorState): { openerLine: number; closerLine: number } | null {
  if (state.doc.lines === 0) return null;
  const FENCE_RE = /^\s*```/;
  const H2_CODE_RE = /^\s*##\s+Code\s*$/;
  const H2_ANY_RE  = /^\s*##\s+.+$/;
  let inCodeSection = false;
  const total = state.doc.lines;
  for (let i = 1; i <= total; i++) {
    const text = state.doc.line(i).text;
    if (H2_CODE_RE.test(text)) { inCodeSection = true; continue; }
    if (H2_ANY_RE.test(text))  { inCodeSection = false; continue; }
    if (inCodeSection && FENCE_RE.test(text)) {
      for (let j = i + 1; j <= total; j++) {
        if (FENCE_RE.test(state.doc.line(j).text)) return { openerLine: i, closerLine: j };
      }
      return null;  // unterminated
    }
  }
  return null;
}
```

**Behavioral parity contract with analog**: first fence under `## Code` wins; fences outside `## Code` are ignored; unterminated fence returns null. Matches analog's "first `<pre>` under `## Code`" behavior (RESEARCH Pitfall 9).

export function buildCodeActionsEditorExtension(plugin: PluginHost) {
  return StateField.define<DecorationSet>({
    create(state) { return buildDecorations(state, plugin); },
    update(old, tr) {

    provide(field) { return EditorView.decorations.from(field); },
  });
}
