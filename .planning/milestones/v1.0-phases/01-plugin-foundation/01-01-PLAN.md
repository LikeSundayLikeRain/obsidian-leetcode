This plan's imports of those types are forward references; all four target modules are created by Plans 02/03/05, and the field declarations use definite-assignment assertions (`!:`) so no value is required at construction time.

Output: A buildable, lintable, testable plugin repo with zero source behavior — just scaffolding, types, redacting utilities, and a field-stubbed `main.ts` whose `onload()` is intentionally empty.

    **`src/shared/logger.ts`** (per AUTH-06, CF-03, RESEARCH.md Pitfall 5 — mandatory redaction):
    ```typescript
    /**
     * Logger that redacts any object key matching /session|csrf|cookie|token/i.
     * MUST be used for any contextual logging that might carry auth objects.
     * Direct `console.*` calls with auth values are forbidden (enforced by grep gate).
     */
    const REDACT = /session|csrf|cookie|token/i;

    function redact(obj: unknown): unknown {
      if (!obj || typeof obj !== 'object') return obj;
      const out: Record<string, unknown> = {};
      for (const [k, v] of Object.entries(obj as Record<string, unknown>)) {
        out[k] = REDACT.test(k) ? '[REDACTED]' : v;
      }
      return out;
    }

    export const logger = {
      debug: (msg: string, ctx?: unknown): void => {
        console.debug(`[leetcode] ${msg}`, ctx !== undefined ? redact(ctx) : '');
      },
      info: (msg: string, ctx?: unknown): void => {
        console.info(`[leetcode] ${msg}`, ctx !== undefined ? redact(ctx) : '');
      },
      warn: (msg: string, ctx?: unknown): void => {
    ```
