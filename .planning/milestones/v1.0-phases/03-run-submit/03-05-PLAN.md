    // src/solve/pollingLoop.ts
    //
    // Phase 3 — exponential-backoff polling loop (D-21, D-22, D-23, D-26).
    // Pure-logic module: takes a `check` async function, an abort flag, and a
    // plugin.registerInterval passthrough; returns the first terminal
    // (`state: 'SUCCESS'`) response or rejects.
    //
    // Cadence: 1s → 2s → 4s → 8s → 8s → 8s ... (last value held).
    // Cap: 60s wall-clock. 3 consecutive errors → treat as judge timeout.
    //
    // Timer registration: every scheduled setWindowTimeout handle is passed to
    // args.registerInterval so Obsidian cancels the timer on plugin unload
    // (prevents orphaned polls after disable/reload). The orchestrator binds
    // `plugin.registerInterval.bind(plugin)` before passing in.
    //
    // Three-point abort check per Pitfall 4:
    //   1. Before scheduling next iteration.
    //   2. Entry to timer callback.
    //   3. After the `await check()` resolves / rejects.

    import { setWindowTimeout, clearWindowTimeout, type TimerHandle } from '../shared/timers';
    import { AbortError, JudgeTimeoutError } from '../shared/errors';
    import { isTerminal, type CheckResponse, type RunCheckResponse, type SubmitCheckResponse } from './types';

    export const BACKOFF_MS = [1000, 2000, 4000, 8000] as const;
    export const MAX_WALLCLOCK_MS = 60_000;
    export const MAX_CONSECUTIVE_ERRORS = 3;

    export interface PollArgs {
      check: () => Promise<CheckResponse>;
      abortFlag: { cancelled: boolean };
      /**
       * Plugin-scoped timer registration. Orchestrator passes
       * `plugin.registerInterval.bind(plugin)` so Obsidian cancels timers on
       * plugin unload. Without this, chained setTimeouts leak across reloads.
       */
      registerInterval: (handle: number) => void;
      /** Optional observer used by tests to confirm cadence. */
      onTick?: (attempt: number, intervalMs: number, cumulativeMs: number) => void;
    }

    export type TerminalResponse = RunCheckResponse | SubmitCheckResponse;

    export function pollUntilTerminal(args: PollArgs): Promise<TerminalResponse> {
      const { check, abortFlag, registerInterval, onTick } = args;
      const startedAt = Date.now();
      let attempt = 0;
      let consecutiveErrors = 0;
      let activeHandle: TimerHandle | null = null;

      return new Promise<TerminalResponse>((resolve, reject) => {
        const scheduleNext = (): void => {
          // Abort check #1 — before scheduling.
          if (abortFlag.cancelled) {
            if (activeHandle) clearWindowTimeout(activeHandle);
            return reject(new AbortError());
          }
          const elapsed = Date.now() - startedAt;
          if (elapsed >= MAX_WALLCLOCK_MS) {
            return reject(new JudgeTimeoutError());
          }
          const intervalMs = BACKOFF_MS[Math.min(attempt, BACKOFF_MS.length - 1)] as number;
          onTick?.(attempt, intervalMs, elapsed + intervalMs);
          attempt++;
          activeHandle = setWindowTimeout(() => {
            activeHandle = null;
            // Abort check #2 — entry to timer callback.
            if (abortFlag.cancelled) return reject(new AbortError());
            void (async () => {
              try {
                const res = await check();
                // Abort check #3 — after await (Pitfall 4 critical point).
                if (abortFlag.cancelled) return reject(new AbortError());
                consecutiveErrors = 0;
                if (isTerminal(res)) {
                  return resolve(res);
                }
                // state is PENDING/STARTED; re-schedule.
                scheduleNext();
              } catch (err) {
                // SessionExpiredError bypasses the retry loop — it's a terminal-fail
                // condition, not a transient error.
                if ((err as Error).name === 'SessionExpiredError') {
                  return reject(err);
                }
                consecutiveErrors++;
                if (consecutiveErrors >= MAX_CONSECUTIVE_ERRORS) {
                  return reject(new JudgeTimeoutError());
                }
                scheduleNext();
              }
            })();
          }, intervalMs);
          // Warning 7 fix — register every scheduled timer handle so Obsidian
          // cancels it on plugin unload.
          registerInterval(activeHandle as unknown as number);
        };

        scheduleNext();
      });
    }
    ```

    Key design points:
    - `registerInterval` is a REQUIRED arg (not optional). Orchestrator passes `plugin.registerInterval.bind(plugin)`.
    - Every `setWindowTimeout(...)` return value is passed through `registerInterval(handle as unknown as number)` immediately.
    - `consecutiveErrors` resets to 0 on any successful response (even PENDING), per D-26 ("3 **consecutive** non-2xx").
    - `SessionExpiredError` bypasses retry — surfaces to orchestrator as terminal fail.
    - `activeHandle` cleanup on the pre-schedule abort prevents orphan timers.
    - `onTick` callback lets tests assert cumulative timestamps without reaching into internals.

    Test adaptation: tests construct a no-op `registerInterval: () => {}` stub when the plugin framework isn't present. Production orchestrator passes the real binding.

    // src/solve/SubmissionOrchestrator.ts
    //
    // Phase 3 — singleton orchestrator. Owns single-flight state, abort flag,
    // and the VerdictModal lifecycle. Dispatches to leetcodeRest + pollingLoop.
    //
    // CONTEXT: D-04, D-10, D-12, D-15, D-22, D-23, D-24, D-27.
    // SOLVE-03, SOLVE-05, SOLVE-06, SOLVE-09.
    // Pattern analog: src/notes/NoteWriter.ts (structural DI + session-expiry dispatch).
    //
    // Timer registration: `plugin.registerInterval` is bound once in the
    // constructor and passed through to `pollUntilTerminal` on every poll.
    // Obsidian cancels registered timers on plugin unload — without this,
    // chained setTimeouts leak.

    import { Notice, MarkdownView, type App, type Plugin, type TFile } from 'obsidian';
    import { interpretSolution, submitSolution, checkSubmission } from './leetcodeRest';
    import { extractFirstFencedBlock } from './codeExtractor';
    import { resolveLangSlug } from './languages';
    import { pollUntilTerminal, type TerminalResponse } from './pollingLoop';
    import type { SubmissionContext, CheckResponse } from './types';
    import {
      SessionExpiredError,
      NoCodeBlockError,
      InProgressError,
      AbortError,
      JudgeTimeoutError,
      UnknownVerdictError,
    } from '../shared/errors';
    import { classifyStatus } from './statusMap';
    import { logger } from '../shared/logger';
    import type { AuthCookies, DetailCacheEntry } from '../settings/SettingsStore';

    // ── Dependency shapes (structural DI for testability) ────────────────

    export interface OrchestratorSettings {
      getAuthCookies(): AuthCookies | null;
      getDefaultLanguage(): string;
      getProblemDetail(slug: string): DetailCacheEntry | null;
      getInternalQuestionId(slug: string): string | null;
    }

    export interface VerdictModalHandle {
      renderPending(): void;
      renderVerdict(res: TerminalResponse, problemTitle: string): void;
      renderUnknown(payload: unknown, problemTitle: string): void;
      renderTimeout(): void;
      close(): void;
    }

    export interface VerdictModalFactoryArgs {
      problemTitle: string;
      onCancel: () => void;
    }

    export type VerdictModalFactory = (args: VerdictModalFactoryArgs) => VerdictModalHandle;

    export interface OrchestratorDeps {
      app: App;
      plugin: Plugin;
      settings: OrchestratorSettings;
      modalFactory: VerdictModalFactory;
    }

    // ── Orchestrator ──────────────────────────────────────────────────────

    export class SubmissionOrchestrator {
      private inFlight = false;
      private abortFlag: { cancelled: boolean } | null = null;
      private activeModal: VerdictModalHandle | null = null;
      private readonly registerInterval: (handle: number) => void;

      constructor(private readonly deps: OrchestratorDeps) {
        // Bind plugin.registerInterval ONCE so polling loops can register timers
        // for plugin-unload cleanup (Warning 7). Binding preserves `this`.
        this.registerInterval = (handle: number): void => {
          deps.plugin.registerInterval(handle);
        };
      }

      isInFlight(): boolean {
        return this.inFlight;
      }

      cancel(): void {
        if (!this.inFlight || !this.abortFlag) return;
        this.abortFlag.cancelled = true;
        // Modal is closed by the pending-state Cancel button click handler;
        // here we only flip the flag in case cancel was called programmatically.
        this.activeModal?.close();
      }

      async submit(): Promise<void> {
        return this.runFlow({ kind: 'submit' });
      }

      async runSample(): Promise<void> {
        const ctx = this.buildContextOrAbort();
        if (!ctx) return;
        const detail = this.deps.settings.getProblemDetail(ctx.slug);
        const sample = detail?.exampleTestcases ?? '';
        return this.runFlow({ kind: 'interpret', dataInput: sample }, ctx);
      }

      async runCustom(dataInput: string): Promise<void> {
        return this.runFlow({ kind: 'interpret', dataInput });
      }

      // ── Internal ────────────────────────────────────────────────────────

      private async runFlow(
        op: { kind: 'submit' } | { kind: 'interpret'; dataInput: string },
        preComputedCtx?: SubmissionContext,
      ): Promise<void> {
        // Gate 1: single-flight (D-24).
        if (this.inFlight) {
          // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
          new Notice('A submission is already in progress. Cancel it first or wait for the verdict.', 6000);
          return;
        }

        // Gate 2: build context (active view, extract code, resolve lang, resolve questionId).
        const ctx = preComputedCtx ?? this.buildContextOrAbort();
        if (!ctx) return;

        // Gate 3: cookies available.
        const cookies = this.deps.settings.getAuthCookies();
        if (!cookies) {
          // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
          new Notice('LeetCode session expired. Log in again.', 8000);
          return;
        }

        // Enter in-flight.
        this.inFlight = true;
        this.abortFlag = { cancelled: false };
        const abortFlag = this.abortFlag;

        // Open pending-state modal.
        this.activeModal = this.deps.modalFactory({
          problemTitle: ctx.problemTitle,
          onCancel: () => {
            abortFlag.cancelled = true;
            // Modal closes itself; orchestrator finalizes state below when
            // pollUntilTerminal rejects with AbortError.
          },
        });
        this.activeModal.renderPending();

        try {
          const id = await this.dispatchStart(op, ctx, cookies);
          if (abortFlag.cancelled) throw new AbortError();
          const terminal = await pollUntilTerminal({
            check: () => checkSubmission({
              id,
              slug: ctx.slug,
              // Read cookies per call (Pitfall 2 — no caching).
              cookies: this.deps.settings.getAuthCookies() ?? cookies,
            }),
            abortFlag,
            // Warning 7 fix — propagate plugin.registerInterval so every
            // scheduled poll timer is cancellable on plugin unload.
            registerInterval: this.registerInterval,
          });
          // Classify status.
          const info = classifyStatus(terminal.status_code, terminal.status_msg);
          if (info.kind === 'unknown') {
            logger.warn('solve.verdict: unrecognized status', {
              status_code: terminal.status_code,
              status_msg: terminal.status_msg,
              payload: terminal,
            });
            this.activeModal?.renderUnknown(terminal, ctx.problemTitle);
            return;
          }
          this.activeModal?.renderVerdict(terminal, ctx.problemTitle);
        } catch (err) {
          this.handleError(err, ctx);
        } finally {
          this.inFlight = false;
          this.abortFlag = null;
          // Do NOT close activeModal in finally — the modal stays open for the
          // user to read the verdict. The user clicks Close / Cancel / presses Escape.
          // Only the abort / timeout / session-expiry branches close the modal.
          if (this.wasAbortedOrTimedOut) {
            this.activeModal?.close();
            this.activeModal = null;
            this.wasAbortedOrTimedOut = false;
          }
        }
      }

      private wasAbortedOrTimedOut = false;

      private async dispatchStart(
        op: { kind: 'submit' } | { kind: 'interpret'; dataInput: string },
        ctx: SubmissionContext,
        cookies: AuthCookies,
      ): Promise<string> {
        if (op.kind === 'submit') {
          const res = await submitSolution({
            slug: ctx.slug,
            lang: ctx.langSlug,
            questionId: ctx.questionId,
            typedCode: ctx.typedCode,
            cookies,
          });
          return res.submission_id;
        }
        const res = await interpretSolution({
          slug: ctx.slug,
          lang: ctx.langSlug,
          questionId: ctx.questionId,
          typedCode: ctx.typedCode,
          dataInput: op.dataInput,
          cookies,
        });
        return res.interpret_id;
      }

      private handleError(err: unknown, _ctx: SubmissionContext): void {
        if (err instanceof AbortError) {
          // User clicked Cancel — silent (no Notice).
          this.wasAbortedOrTimedOut = true;
          return;
        }
        if (err instanceof JudgeTimeoutError) {
          this.activeModal?.renderTimeout();
          // Modal displays the timeout state; user closes manually.
          return;
        }
        if (err instanceof SessionExpiredError) {
          this.wasAbortedOrTimedOut = true;
          // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
          new Notice('LeetCode session expired. Log in again.', 8000);
          return;
        }
        if (err instanceof UnknownVerdictError) {
          // Rare — thrown by leetcodeRest in exotic cases.
          this.activeModal?.renderUnknown(err.payload, _ctx.problemTitle);
          return;
        }
        // Generic network or parse error.
        logger.warn('solve.submit: unexpected error', err);
        this.activeModal?.close();
        this.activeModal = null;
        // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
        new Notice(`LeetCode submission failed. Try again.`, 6000);
      }

      /**
       * Validates active view + extracts code + builds SubmissionContext.
       * Returns null and fires a Notice if any gate fails.
       */
      private buildContextOrAbort(): SubmissionContext | null {
        const view = this.deps.app.workspace.getActiveViewOfType(MarkdownView);
        if (!view || !view.file) {
          // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
          new Notice('Open a LeetCode problem note first.', 4000);
          return null;
        }
        const file = view.file;
        const cache = this.deps.app.metadataCache.getFileCache(file);
        const fm = cache?.frontmatter as Record<string, unknown> | undefined;
        const slug = fm?.['lc-slug'];
        const title = fm?.['lc-title'];
        if (typeof slug !== 'string' || !slug) {
          // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
          new Notice('Open a LeetCode problem note first.', 4000);
          return null;
        }

        // Read note body at invocation (SOLVE-09 current-content-at-submit).
        const body = view.editor.getValue();
        const extracted = extractFirstFencedBlock(body);
        if (!extracted) {
          // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
          new Notice('No code block found. Add a fenced block with your solution.', 6000);
          return null;
        }

        const defaultLang = this.deps.settings.getDefaultLanguage();
        const langSlug = resolveLangSlug(extracted.lang, defaultLang);

        // Resolve internal questionId (D-30).
        const internalId = this.deps.settings.getInternalQuestionId(slug);
        if (!internalId) {
          // Fallback: try frontmatter lc-id (questionFrontendId) — works for
          // most problems; premium variants may diverge. Log warn.
          const fmId = fm?.['lc-id'];
          if (typeof fmId !== 'string' && typeof fmId !== 'number') {
            logger.warn('solve.ctx: missing internalQuestionId and lc-id', { slug });
            // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
            new Notice('LeetCode problem id missing. Reopen the problem first.', 4000);
            return null;
          }
          logger.warn('solve.ctx: using lc-id as question_id fallback', { slug });
          return {
            slug,
            langSlug,
            typedCode: extracted.code,
            questionId: String(fmId),
            problemTitle: typeof title === 'string' ? title : slug,
          };
        }

        return {
          slug,
          langSlug,
          typedCode: extracted.code,
          questionId: internalId,
          problemTitle: typeof title === 'string' ? title : slug,
        };
      }
    }
    ```
