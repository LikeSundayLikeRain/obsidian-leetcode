    ```typescript
    // src/solve/customTestStore.ts
    //
    // Vault wrappers around pure CaseRegion transforms. Enforces CF-06:
    // no vault.modify; only vault.process + vault.read.

    import type { App, TFile } from 'obsidian';
    import { readCases, writeCases } from './CaseRegion';
    import { logger } from '../shared/logger';

    export async function readCasesFromVault(app: App, file: TFile): Promise<string[]> {
      try {
        const body = await app.vault.read(file);
        return readCases(body);
      } catch (err) {
        logger.debug('solve.customTestStore.read: non-fatal failure', err);
        return [];
      }
    }

    export async function writeCasesToVault(
      app: App,
      file: TFile,
      cases: string[],
    ): Promise<void> {
      await app.vault.process(file, (current) => writeCases(current, cases));
    }
    ```

    That's the whole module. Tests assert the round-trip: `write([a, b, c])` then `read()` returns `[a, b, c]`.

    Do NOT catch errors in `writeCasesToVault` — caller (CustomTestModal.onClose in Task 3) handles failure with a silent debug log or Notice.

    ```typescript
    // src/solve/VerdictModal.ts
    //
    // Phase 3 verdict-result modal — 8 render states:
    //   pending · AC · WA · TLE · MLE · CE · RE · unknown · timeout
    //
    // Spec: .planning/phases/03-run-submit/03-UI-SPEC.md
    // Pattern: Modal refs + mutation (see RESEARCH.md Pattern 3).
    // CF-07: all DOM via createEl / createDiv / createSpan + setText.

    import { Modal, Notice, setIcon, type App } from 'obsidian';
    import type { RunCheckResponse, SubmitCheckResponse } from './types';
    import { classifyStatus, type VerdictKind } from './statusMap';
    import { logger } from '../shared/logger';

    type TerminalResponse = RunCheckResponse | SubmitCheckResponse;

    export interface VerdictModalArgs {
      problemTitle: string;
      onCancel: () => void;
      /** Called when user clicks "Copy failing testcase to custom input" on WA/TLE/RE. */
      onCopyFailingInput?: (input: string) => void;
    }

    const PAYLOAD_DISPLAY_MAX = 2048;

    export class VerdictModal extends Modal {
      private runtimeRowEl!: HTMLElement;
      private bodyEl!: HTMLElement;
      private footerEl!: HTMLElement;
      private readonly args: VerdictModalArgs;

      constructor(app: App, args: VerdictModalArgs) {
        super(app);
        this.args = args;
      }

      onOpen(): void {
        const { contentEl } = this;
        contentEl.empty();
        contentEl.addClass('leetcode-verdict');

        this.runtimeRowEl = contentEl.createDiv({ cls: 'leetcode-verdict-runtime' });
        this.bodyEl = contentEl.createDiv({ cls: 'leetcode-verdict-body' });
        this.bodyEl.setAttr('aria-live', 'polite');
        this.footerEl = contentEl.createDiv({ cls: 'leetcode-verdict-footer leetcode-verdict-action-row' });

        this.renderPending();
      }

      onClose(): void {
        this.contentEl.empty();
      }

      // ── State transitions (implements VerdictModalHandle from Plan 05) ────

      renderPending(): void {
        this.titleEl.setText('Running…');
        this.contentEl.removeClass('leetcode-verdict-ac', 'leetcode-verdict-wa', 'leetcode-verdict-tle',
          'leetcode-verdict-mle', 'leetcode-verdict-ce', 'leetcode-verdict-re', 'leetcode-verdict-unknown');
        this.contentEl.addClass('leetcode-verdict-pending');
        this.runtimeRowEl.empty();
        this.bodyEl.empty();
        const spinnerWrap = this.bodyEl.createDiv({ cls: 'leetcode-verdict-spinner' });
        setIcon(spinnerWrap, 'loader');
        const primary = this.bodyEl.createEl('p');
        primary.setText('Polling LeetCode for verdict…');
        const subtitle = this.bodyEl.createEl('p', { cls: 'leetcode-verdict-subtitle' });
        subtitle.setText('Backoff: 1s → 2s → 4s → 8s');
        this.footerEl.empty();
        const cancelBtn = this.footerEl.createEl('button');
        cancelBtn.setText('Cancel');
        cancelBtn.addEventListener('click', () => {
          this.args.onCancel();
          this.close();
        });
        cancelBtn.focus();
      }

      renderVerdict(res: TerminalResponse, problemTitle: string): void {
        const info = classifyStatus(res.status_code, res.status_msg);
        this.contentEl.removeClass('leetcode-verdict-pending');
        this.contentEl.addClass(`leetcode-verdict-${info.kind === 'unknown-lc' ? 'unknown' : info.kind}`);
        this.titleEl.setText(`${info.displayName} — ${problemTitle}`);
        this.renderRuntimeRow(res);
        this.bodyEl.empty();
        this.footerEl.empty();
        switch (info.kind) {
          case 'ac': this.renderAc(res); break;
          case 'wa': this.renderWa(res); break;
          case 'tle': this.renderTle(res); break;
          case 'mle': this.renderMle(res); break;
          case 'ce': this.renderCe(res); break;
          case 're': this.renderRe(res); break;
          case 'ie':
          case 'ole':
          case 'unknown-lc':
            // Treat these as visual-error with close-only footer.
            this.renderGenericError(res, info.displayName);
            break;
          case 'unknown':
            this.renderUnknown(res, problemTitle);
            return; // renderUnknown builds its own footer
        }
        this.addCloseButton(true);
      }

      renderUnknown(payload: unknown, problemTitle: string): void {
        this.contentEl.removeClass('leetcode-verdict-pending');
        this.contentEl.addClass('leetcode-verdict-unknown');
        this.titleEl.setText(`Unrecognized verdict — ${problemTitle}`);
        this.runtimeRowEl.empty();
        this.bodyEl.empty();

        const intro = this.bodyEl.createEl('p');
        intro.setText('LeetCode returned an unrecognized status. Copy the payload to file a bug report.');

        const details = this.bodyEl.createEl('details', { cls: 'leetcode-verdict-unknown-details' });
        const summary = details.createEl('summary');
        summary.setText('Raw response');
        const pre = details.createEl('pre');
        const payloadText = safeStringify(payload).slice(0, PAYLOAD_DISPLAY_MAX);
        pre.setText(payloadText);

        this.footerEl.empty();
        const copyBtn = this.footerEl.createEl('button');
        copyBtn.setText('Copy payload');
        copyBtn.addEventListener('click', () => {
          const redacted = typeof (logger as unknown as { redact?: (v: unknown) => unknown }).redact === 'function'
            ? (logger as unknown as { redact: (v: unknown) => unknown }).redact(payload)
            : payload;
          void navigator.clipboard.writeText(safeStringify(redacted));
          // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
          new Notice('Payload copied.', 2000);
        });
        this.addCloseButton(false);
      }

      renderTimeout(): void {
        this.contentEl.removeClass('leetcode-verdict-pending');
        this.titleEl.setText('Judge timeout');
        this.runtimeRowEl.empty();
        this.bodyEl.empty();
        const p = this.bodyEl.createEl('p');
        p.setText('LeetCode judge timed out. Try again or check leetcode.com.');
        this.footerEl.empty();
        const closeBtn = this.footerEl.createEl('button', { cls: 'mod-cta' });
        closeBtn.setText('Close');
        closeBtn.addEventListener('click', () => this.close());
        closeBtn.focus();
      }

      // ── Per-verdict body renderers ────────────────────────────────────────

      private renderRuntimeRow(res: TerminalResponse): void {
        this.runtimeRowEl.empty();
        const runtime = (res as SubmitCheckResponse).status_runtime ?? '—';
        const memory = (res as SubmitCheckResponse).status_memory ?? '—';
        if (runtime !== '—' || memory !== '—') {
          this.runtimeRowEl.setText(`Runtime: ${runtime} · Memory: ${memory}`);
        }
      }

      private renderAc(res: TerminalResponse): void {
        const display = this.bodyEl.createDiv({ cls: 'leetcode-verdict-ac-display' });
        display.setText('Accepted');
        const submit = res as SubmitCheckResponse;
        if (typeof submit.runtime_percentile === 'number' || typeof submit.memory_percentile === 'number') {
          const pct = this.bodyEl.createDiv({ cls: 'leetcode-verdict-percentile' });
          const rt = submit.runtime_percentile?.toFixed(1) ?? '—';
          const mem = submit.memory_percentile?.toFixed(1) ?? '—';
          pct.setText(`Beats ${rt}% (runtime) · ${mem}% (memory)`);
        }
      }

      private renderWa(res: TerminalResponse): void {
        const submit = res as SubmitCheckResponse;
        const runField = res as RunCheckResponse;
        const input = submit.input ?? submit.last_testcase ?? ''; // prefer submit-shape input
        const output = submit.std_output ?? firstString(runField.code_output) ?? firstString(runField.code_answer) ?? '';
        const expected = submit.expected_output ?? firstString(runField.expected_code_answer) ?? '';

        this.makeDiffSection('Input', input, 'leetcode-verdict-diff-input');
        this.makeDiffSection('Output', output, 'leetcode-verdict-diff-actual', 'leetcode-verdict-section-label--output');
        this.makeDiffSection('Expected', expected, 'leetcode-verdict-diff-expected', 'leetcode-verdict-section-label--expected');

        this.addCopyFailingInputButton(input);
      }

      private renderTle(res: TerminalResponse): void {
        const submit = res as SubmitCheckResponse;
        const input = submit.input ?? submit.last_testcase ?? '';
        this.makeDiffSection('Input', input, 'leetcode-verdict-diff-input');
        const runField = res as RunCheckResponse;
        const lastOut = firstString(runField.code_output);
        if (lastOut) {
          this.makeDiffSection('Last output', lastOut, 'leetcode-verdict-diff-input');
        }
        this.addCopyFailingInputButton(input);
      }

      private renderMle(res: TerminalResponse): void {
        const submit = res as SubmitCheckResponse;
        const input = submit.input ?? submit.last_testcase ?? '';
        this.makeDiffSection('Input', input, 'leetcode-verdict-diff-input');
        // MLE has NO action button per UI-SPEC § action buttons table
      }

      private renderCe(res: TerminalResponse): void {
        const submit = res as SubmitCheckResponse;
        const errText = submit.full_compile_error ?? submit.compile_error ?? '';
        const label = this.bodyEl.createDiv({ cls: 'leetcode-verdict-section-label' });
        label.setText('Compile error');
        const pre = this.bodyEl.createEl('pre', { cls: 'leetcode-verdict-error-pre' });
        pre.setText(errText);
        const copyBtn = this.footerEl.createEl('button');
        copyBtn.setText('Copy error');
        copyBtn.addEventListener('click', () => {
          void navigator.clipboard.writeText(errText);
          // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
          new Notice('Error copied.', 2000);
        });
      }

      private renderRe(res: TerminalResponse): void {
        const submit = res as SubmitCheckResponse;
        const errText = submit.full_runtime_error ?? submit.runtime_error ?? '';
        const label = this.bodyEl.createDiv({ cls: 'leetcode-verdict-section-label' });
        label.setText('Error');
        const pre = this.bodyEl.createEl('pre', { cls: 'leetcode-verdict-error-pre' });
        pre.setText(errText);
        const input = submit.input ?? submit.last_testcase ?? '';
        if (input) {
          this.makeDiffSection('Input', input, 'leetcode-verdict-diff-input');
        }
        this.addCopyFailingInputButton(input);
      }

      private renderGenericError(res: TerminalResponse, label: string): void {
        const p = this.bodyEl.createEl('p');
        p.setText(label);
      }

      private makeDiffSection(labelText: string, value: string, preCls: string, labelCls?: string): void {
        const section = this.bodyEl.createDiv({ cls: 'leetcode-verdict-section' });
        const label = section.createDiv({ cls: 'leetcode-verdict-section-label' + (labelCls ? ' ' + labelCls : '') });
        label.setText(labelText);
        label.setAttr('aria-label', labelText);
        const pre = section.createEl('pre', { cls: preCls });
        pre.setText(value);
      }

      private addCopyFailingInputButton(input: string): void {
        if (!this.args.onCopyFailingInput || !input) return;
        const copyBtn = this.footerEl.createEl('button');
        copyBtn.setText('Copy failing testcase to custom input');
        copyBtn.addEventListener('click', () => {
          this.args.onCopyFailingInput?.(input);
          this.close();
        });
      }

      private addCloseButton(primary: boolean): void {
        const closeBtn = this.footerEl.createEl('button', { cls: primary ? 'mod-cta' : undefined });
        closeBtn.setText('Close');
        closeBtn.addEventListener('click', () => this.close());
        closeBtn.focus();
      }
    }

    function safeStringify(v: unknown): string {
      try { return JSON.stringify(v, null, 2); }
      catch { return String(v); }
    }

    function firstString(v: string | string[] | undefined): string | undefined {
      if (Array.isArray(v)) return v[0];
      return v;
    }
    ```

    ```typescript
    // src/solve/CustomTestModal.ts
    //
    // Phase 3 — tabbed custom-input modal (D-17, D-18, D-19, D-20).
    // Persists cases to `## Custom Tests` section on close via customTestStore.
    // Pattern: createEl-based tabs; Obsidian Modal lifecycle (onOpen/onClose).

    import { Modal, type App, type TFile } from 'obsidian';
    import { writeCasesToVault } from './customTestStore';
    import { logger } from '../shared/logger';

    export interface CustomTestModalArgs {
      file: TFile;
      initialCases: string[];
      onRun: (input: string) => void;
    }

    export class CustomTestModal extends Modal {
      private cases: Array<{ input: string }> = [];
      private activeTab = 0;
      private tabsEl!: HTMLElement;
      private textareaEl!: HTMLTextAreaElement;
      private readonly args: CustomTestModalArgs;
      private didPersist = false;

      constructor(app: App, args: CustomTestModalArgs) {
        super(app);
        this.args = args;
        this.cases = args.initialCases.length > 0
          ? args.initialCases.map((input) => ({ input }))
          : [{ input: '' }];
      }

      onOpen(): void {
        const { contentEl } = this;
        contentEl.empty();
        contentEl.addClass('leetcode-custom-test');
        this.titleEl.setText('Custom test input');

        this.tabsEl = contentEl.createDiv({ cls: 'leetcode-custom-test-tabs' });
        this.tabsEl.setAttr('role', 'tablist');
        this.renderTabs();

        this.textareaEl = contentEl.createEl('textarea', {
          cls: 'leetcode-custom-test-textarea',
          attr: { rows: '10', spellcheck: 'false', placeholder: 'Enter test input (one value per line)' },
        });
        this.textareaEl.setAttr('role', 'tabpanel');
        this.textareaEl.value = this.cases[this.activeTab]?.input ?? '';
        this.textareaEl.addEventListener('input', () => {
          const c = this.cases[this.activeTab];
          if (c) c.input = this.textareaEl.value;
        });
        this.textareaEl.focus();

        const footer = contentEl.createDiv({ cls: 'leetcode-custom-test-footer' });
        const runBtn = footer.createEl('button', { cls: 'mod-cta' });
        runBtn.setText('Run');
        runBtn.addEventListener('click', () => {
          this.syncActiveFromTextarea();
          const input = this.cases[this.activeTab]?.input ?? '';
          // Persist first so cases are saved even if onRun throws.
          void this.persist();
          this.args.onRun(input);
          this.close();
        });
      }

      onClose(): void {
        // Persist-on-close (UI-SPEC §Interaction Contract).
        if (!this.didPersist) {
          this.syncActiveFromTextarea();
          void this.persist();
        }
        this.contentEl.empty();
      }

      // ── Tab rendering + actions ───────────────────────────────────────────

      private renderTabs(): void {
        this.tabsEl.empty();
        this.cases.forEach((_, i) => {
          const tab = this.tabsEl.createEl('button', {
            cls: 'leetcode-custom-test-tab' + (i === this.activeTab ? ' is-active' : ''),
          });
          tab.setAttr('role', 'tab');
          tab.setAttr('aria-selected', String(i === this.activeTab));
          tab.setText(`Case ${i + 1}`);
          tab.addEventListener('click', () => this.switchTab(i));
          if (this.cases.length > 1) {
            const remove = tab.createEl('span', { cls: 'leetcode-custom-test-tab-remove' });
            remove.setText('×');
            remove.addEventListener('click', (e) => {
              e.stopPropagation();
              this.removeCase(i);
            });
          }
        });
        const addBtn = this.tabsEl.createEl('button', {
          cls: 'leetcode-custom-test-tab leetcode-custom-test-tab-add',
        });
        addBtn.setText('+');
        addBtn.addEventListener('click', () => this.addCase());
      }

      private switchTab(i: number): void {
        this.syncActiveFromTextarea();
        this.activeTab = i;
        this.textareaEl.value = this.cases[i]?.input ?? '';
        this.textareaEl.focus();
        this.renderTabs();
      }

      private addCase(): void {
        this.syncActiveFromTextarea();
        this.cases.push({ input: '' });
        this.activeTab = this.cases.length - 1;
        this.textareaEl.value = '';
        this.textareaEl.focus();
        this.renderTabs();
      }

      private removeCase(i: number): void {
        if (this.cases.length <= 1) return;
        this.cases.splice(i, 1);
        if (this.activeTab >= this.cases.length) {
          this.activeTab = this.cases.length - 1;
        } else if (this.activeTab > i) {
          this.activeTab--;
        } else if (this.activeTab === i && i > 0) {
          this.activeTab = i - 1;
        }
        this.textareaEl.value = this.cases[this.activeTab]?.input ?? '';
        this.textareaEl.focus();
        this.renderTabs();
      }

      private syncActiveFromTextarea(): void {
        const c = this.cases[this.activeTab];
        if (c) c.input = this.textareaEl.value;
      }

      private async persist(): Promise<void> {
        this.didPersist = true;
        try {
          await writeCasesToVault(this.app, this.args.file, this.cases.map((c) => c.input));
        } catch (err) {
          logger.debug('solve.customTestModal.persist: non-fatal failure', err);
        }
      }
    }
    ```

    Key rules:
    - Tab buttons use `role="tab"` + `aria-selected` (UI-SPEC §Accessibility)
    - Textarea uses `role="tabpanel"` + `spellcheck=false` (UI-SPEC §Custom Test Modal copy)
    - Run button uses `mod-cta` for primary CTA accent (UI-SPEC §Color reserved list)
    - persist-on-close happens before `contentEl.empty()` so state is intact
    - `didPersist` guard prevents double-persist when Run → close() also triggers onClose
    - Errors in persist are swallowed + debug-logged (user action was implicit; no need for Notice)
