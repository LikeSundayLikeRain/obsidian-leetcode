    this.addCommand({
      id: 'leetcode-run-sample',
      name: 'Run code (sample)',  // Obsidian renders this as "LeetCode: Run code (sample)" because plugin display name is "LeetCode"
      editorCheckCallback: (checking, _editor, view) => {
        const file = view.file;
        if (!file) return false;
        const fm = this.app.metadataCache.getFileCache(file)?.frontmatter as Record<string, unknown> | undefined;
        if (typeof fm?.['lc-slug'] !== 'string') return false;
        if (!checking) void this.solve.runSample();
        return true;
      },
    });

    this.addCommand({
      id: 'leetcode-run-custom',
      name: 'Run code (custom input)',
      editorCheckCallback: (checking, _editor, view) => {

        if (typeof fm?.['lc-slug'] !== 'string') return false;

        return true;

    });

    this.addCommand({
      id: 'leetcode-submit',
      name: 'Submit',
      editorCheckCallback: (checking, _editor, view) => {
        const file = view.file;
        if (!file) return false;
        const fm = this.app.metadataCache.getFileCache(file)?.frontmatter as Record<string, unknown> | undefined;
        if (typeof fm?.['lc-slug'] !== 'string') return false;
        if (!checking) void this.solve.submit();
        return true;
      },
    });

    this.addCommand({
      id: 'leetcode-insert-starter-code',
      name: 'Insert starter code',
      editorCheckCallback: (checking, _editor, view) => {
        const file = view.file;
        if (!file) return false;
        const fm = this.app.metadataCache.getFileCache(file)?.frontmatter as Record<string, unknown> | undefined;
