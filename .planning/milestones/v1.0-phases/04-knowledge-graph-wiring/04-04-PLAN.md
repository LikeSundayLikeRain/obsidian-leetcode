```typescript
/**
 * Copy-to-## Code helper (D-04, GRAPH-01 revised).
 *
 * Two exports:
 *  - copyToCode(app, file, code, langSlug): atomic vault.process rewrite of
 *    the first fenced block under ## Code. New block's language tag = langSlug
 *    (NOT the old block's tag — D-04 explicit).
 *  - hasExistingCodeBlock(body): predicate used by SubmissionDetailModal to
 *    decide whether to open ConfirmOverwriteModal before rewriting.
 *
 * No new vault-mutation primitive — reuses Phase 3's forceInjectCodeSection
 * pure transform (EXACT shape already solves "rewrite the first fenced
 * block under ## Code with given code + langSlug").
 */
import type { App, TFile } from 'obsidian';
import { forceInjectCodeSection } from '../solve/starterCodeInjector';
import { extractFirstFencedBlock } from '../solve/codeExtractor';

/**
 * Atomic rewrite of the ## Code fenced block with the submitted code + lang.
 * - Uses vault.process (CF-06).
 * - Uses Phase 3's forceInjectCodeSection (pure, idempotent, safe inside retry).
 */
export async function copyToCode(
  app: App,
  file: TFile,
  code: string,
  langSlug: string,
): Promise<void> {
  await app.vault.process(file, (current) =>
    forceInjectCodeSection(current, { starterCode: code, langSlug }),
  );
}

/**
 * Predicate: does the body have a non-empty fenced code block?
 * Used by SubmissionDetailModal to decide whether to open ConfirmOverwriteModal
 * before rewriting. Whitespace-only fences count as empty (no destructive
 * overwrite risk; user confirmation would be annoying per UI-SPEC).
 */
export function hasExistingCodeBlock(body: string): boolean {
  const extracted = extractFirstFencedBlock(body);
  return extracted !== null && extracted.code.trim().length > 0;
}
```

```typescript
export class SubmissionDetailModal extends Modal {
  private readonly args: SubmissionDetailModalArgs;
  private renderChild: Component | null = null;
  private detail: SubmissionDetail | null = null;

  constructor(app: App, args: SubmissionDetailModalArgs) {
    super(app);
    this.args = args;
  }

  override onOpen(): void {
    const { contentEl, titleEl } = this;
    clear(contentEl);
    addClass(contentEl, 'leetcode-submissions-detail');
    if (titleEl) titleEl.textContent = 'Submission';  // updated on detail load
    this.renderLoading();
    void this.loadAndRender();
  }

  override onClose(): void {
    // Pitfall 7 — dispose MarkdownRenderer child to prevent CM6 leak.
    if (this.renderChild) {
      try { this.removeChild(this.renderChild); } catch { /* headless */ }
      this.renderChild = null;
    }
    clear(this.contentEl);
  }

  // ---- private ----

  private async loadAndRender(): Promise<void> {
    try {
      this.detail = await this.args.fetchDetail();
      this.renderDetail(this.detail);
    } catch (err) {
      if (err instanceof SessionExpiredError) {
        // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
        new Notice('LeetCode session expired. Log in again.', 8000);
        this.close();
        return;
      }
      logger.debug('graph.detail: load failed', err);
      this.renderError();
    }
  }

  private renderLoading(): void {
    clear(this.contentEl);
    const box = appendEl(this.contentEl, 'div', 'leetcode-submissions-detail__loading');
    const spin = appendEl(box, 'div', 'leetcode-submissions-detail__spinner');
    setIcon(spin, 'loader');
    const label = appendEl(box, 'div');
    label.textContent = 'Loading submission…';
  }

  private renderError(): void {
    clear(this.contentEl);
    const box = appendEl(this.contentEl, 'div', 'leetcode-submissions-detail__error');
    const h = appendEl(box, 'div', 'leetcode-submissions-detail__error-heading');
    h.textContent = "Couldn't load submission.";
    const b = appendEl(box, 'div', 'leetcode-submissions-detail__error-body');
    b.textContent = 'Check your connection and try again.';
    const retry = appendEl(box, 'button', 'leetcode-submissions-detail__retry') as HTMLButtonElement;
    retry.type = 'button';
    setIcon(retry, 'refresh-cw');
    const rlabel = appendEl(retry, 'span');
    rlabel.textContent = ' Retry';
    retry.addEventListener('click', () => { this.renderLoading(); void this.loadAndRender(); });
  }

  private renderDetail(d: SubmissionDetail): void {
    clear(this.contentEl);

    // Title update — CONTEXT D-04 format '{verdict} · {problem title}' (middle-dot `·` U+00B7,
    // single space each side). Do NOT use em-dash (U+2014) — that's Phase 3's VerdictModal
    // convention; Phase 4 follows CONTEXT D-04 verbatim.
    const { titleEl } = this;
    const info = classifyStatus(d.status_code, d.status_msg);
    const verdictName = info.kind === 'unknown' ? 'Submission' : info.displayName;
    if (titleEl) titleEl.textContent = `${verdictName} · ${this.args.problemTitle}`;

    // Metadata row — UI-SPEC §Layout.
    const meta = appendEl(this.contentEl, 'div', 'leetcode-submissions-detail__meta');
    const runtime = d.runtime && d.runtime !== 'N/A' ? d.runtime : '—';
    const memory = d.memory && d.memory !== 'N/A' ? d.memory : '—';
    const submittedAt = typeof d.submitted_at === 'number' ? formatIsoLocalTz(d.submitted_at) : '';
    meta.textContent = `Runtime: ${runtime} · Memory: ${memory} · ${d.lang}${submittedAt ? ' · ' + submittedAt : ''}`;

    // Code block — MarkdownRenderer.render with Component lifecycle (Pitfall 7).
    const wrap = appendEl(this.contentEl, 'div', 'leetcode-submissions-detail__code');
    const fenced = '```' + d.lang + '\n' + d.code + '\n```';
    this.renderChild = new Component();
    this.addChild(this.renderChild);
    void MarkdownRenderer.render(this.app, fenced, wrap, '', this.renderChild);

    // Footer — Copy-to-Code (accent) + Close (default focus).
    const footer = appendEl(this.contentEl, 'div', 'leetcode-submissions-detail__footer');
    const copyBtn = appendEl(footer, 'button', 'mod-cta leetcode-submissions-detail__copy') as HTMLButtonElement;
    copyBtn.type = 'button';
    setIcon(copyBtn, 'copy');
    const copyLabel = appendEl(copyBtn, 'span');
    copyLabel.textContent = ' Copy to ## Code';
    copyBtn.addEventListener('click', () => { void this.handleCopyToCode(d); });

    const closeBtn = appendEl(footer, 'button', 'leetcode-submissions-detail__close') as HTMLButtonElement;
    closeBtn.type = 'button';
    closeBtn.textContent = 'Close';
    closeBtn.addEventListener('click', () => this.close());

    // UI-SPEC §Accessibility — default focus on Close.
    try { closeBtn.focus(); } catch { /* headless */ }
  }

  private async handleCopyToCode(detail: SubmissionDetail): Promise<void> {
    const currentBody = await this.app.vault.read(this.args.file);
    if (hasExistingCodeBlock(currentBody)) {
      new ConfirmOverwriteModal(this.app, {
        onConfirm: async () => {
          await copyToCode(this.app, this.args.file, detail.code, detail.lang);
          this.close();
        },
      }).open();
    } else {
      await copyToCode(this.app, this.args.file, detail.code, detail.lang);
      this.close();
    }
  }
}
```
