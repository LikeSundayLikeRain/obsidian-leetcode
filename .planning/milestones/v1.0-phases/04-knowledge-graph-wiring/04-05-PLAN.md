**Step 1 — create src/graph/SubmissionHistoryStore.ts (~80 LoC):**

```typescript
/**
 * In-memory per-slug cache of submission-list results (CONTEXT D-02;
 * RESEARCH §Open Questions (RESOLVED) Q1).
 *
 * NOT persisted to data.json (D-07 prohibits). Cleared on plugin unload
 * (store lifetime = plugin lifetime).
 *
 * Two roles:
 *  1. NoteWriter.openProblem fires `refetch(slug, cookies)` after reveal
 *     (fire-and-forget, silent-fail per D-12).
 *  2. SubmissionPickerModal reads `get(slug)` first (instant render if
 *     populated), then kicks its own background refresh to update the UI.
 *
 * Non-goals:
 *  - No TTL (D-02 says always-live — refetch-on-open replaces the last value).
 *  - No per-row detail caching (detail still fetches lazily on picker-row click).
 *  - No listener/subscriber API (the picker polls via its own render loop;
 *    no React-style reactive update).
 */
import { logger } from '../shared/logger';
import { listSubmissionsForSlug } from './submissionHistoryClient';
import type { SubmissionRow } from './submissionHistoryClient';
import type { AuthCookies } from '../settings/SettingsStore';

export class SubmissionHistoryStore {
  private readonly cache = new Map<string, SubmissionRow[]>();

  /** Instant lookup for picker warm-start. Undefined = cache miss. */
  get(slug: string): SubmissionRow[] | undefined {
    return this.cache.get(slug);
  }

  /** Caller-explicit set (used by tests + the picker's own background refresh). */
  set(slug: string, rows: SubmissionRow[]): void {
    this.cache.set(slug, rows);
  }

  /** Clear all entries (used on plugin unload; optional). */
  clear(): void {
    this.cache.clear();
  }

  /**
   * Fire-and-forget refetch (D-02). Silent on every failure path:
   *  - Missing cookies → debug log, no throw
   *  - Network / LC error → debug log, no throw
   *  - SessionExpiredError → debug log (the Notice fires when the USER opens
   *    the picker, not here — this is a background prefetch).
   *
   * Always returns void — callers use fire-and-forget pattern.
   */
  async refetch(slug: string, cookies: AuthCookies | null): Promise<void> {
    if (!cookies) {
      logger.debug('graph.SubmissionHistoryStore.refetch: no cookies, skipping', { slug });
      return;
    }
    try {
      const rows = await listSubmissionsForSlug(slug, cookies);
      this.cache.set(slug, rows);
      logger.debug('graph.SubmissionHistoryStore.refetch: cached', { slug, count: rows.length });
    } catch (err) {
      logger.debug('graph.SubmissionHistoryStore.refetch: swallowed failure', { slug, err });
    }
  }
}
```

```typescript
private async loadAndRender(): Promise<void> {
  // BLOCKER 1 — warm-start from the in-memory store if populated.
  const cached = this.args.submissionHistoryStore?.get(this.args.slug);
  if (cached !== undefined) {
    this.rows = cached;
    if (this.rows.length === 0) this.renderEmpty();
    else this.renderRows(this.rows);
    // Kick a background refresh to update if LC returned new rows.
    void (async () => {
      try {
        const fresh = await this.args.fetchHistory();
        this.args.submissionHistoryStore?.set(this.args.slug, fresh);
        // Only re-render if the row count or ordering actually changed (cheap check).
        if (this.rowsDiffer(this.rows, fresh)) {
          this.rows = fresh;
          if (this.rows.length === 0) this.renderEmpty();
          else this.renderRows(this.rows);
        }
      } catch (err) {
        // Background refresh failure is silent — the warm-start cache is still rendered.
        // (If the user triggers Retry later, the visible error path fires.)
        logger.debug('graph.picker: background refresh failed (warm-start still rendered)', err);
      }
    })();
    return;
  }

  // Cache miss → original fetch-first flow.
  try {
    this.rows = await this.args.fetchHistory();
    this.args.submissionHistoryStore?.set(this.args.slug, this.rows);
    if (this.rows.length === 0) this.renderEmpty();
    else this.renderRows(this.rows);
  } catch (err) {
    if (err instanceof SessionExpiredError) {
      // eslint-disable-next-line obsidianmd/ui/sentence-case -- UI-SPEC LOCKED
      new Notice('LeetCode session expired. Log in again.', 8000);
      this.close();
      return;
    }
    logger.debug('graph.picker: load failed', err);
    this.renderError({ retry: () => { this.renderLoading(); void this.loadAndRender(); } });
  }
}

/** Shallow identity check — true if rows lists differ in length or in the id
 *  of any row at the same index. Good enough to avoid re-rendering when the
 *  user hasn't made new submissions between warm-start and background refresh. */
private rowsDiffer(a: SubmissionRow[], b: SubmissionRow[]): boolean {
  if (a.length !== b.length) return true;
  for (let i = 0; i < a.length; i++) {
    if (a[i].id !== b[i].id) return true;
  }
  return false;
}
```
